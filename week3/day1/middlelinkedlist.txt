class Solution {
    public ListNode middleNode(ListNode head) {
        ListNode temp = head;
        int count = 0;

        while(temp != null){
            count++;
            temp = temp.next;
        }

        int n = count / 2;

        ListNode mid = head;
        for(int i = 0; i < n; i++){
            mid = mid.next;
        }

        return mid;
    }
}
